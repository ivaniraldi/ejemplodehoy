ls = listar directorios
pwd = ver la ruta donde estoy parado en la terminal
ls -a = listar directorios y archivos ocultos
mkdir = crear carpetas(directorios)

git add .

PARA ETIQUETAR LOS ARCHIVOS
git commit -m "mensaje"
